Lemma to_Z_bounded : forall x, (0 <= to_Z x < 9223372036854775808)%Z.

Instance Inj_int_Z : InjTyp int Z :=
  mkinj _ _ to_Z (fun x => 0 <= x < 9223372036854775808)%Z to_Z_bounded.
Add Zify InjTyp Inj_int_Z.
