Module Cc_plugin__Ccproof

type rule =
| Ax of Cc_plugin.Ccalgo.axiom

if ⊢ t = u :: A, then ⊢ t = u :: A

| SymAx of Cc_plugin.Ccalgo.axiom

if ⊢ t = u : A, then ⊢ u = t :: A

| Refl of Cc_plugin.Ccalgo.ATerm.t
| Trans of proof * proof

⊢ t = u :: A -> ⊢ u = v :: A -> ⊢ t = v :: A

| Congr of proof * proof

⊢ f = g :: forall x : A, B -> ⊢ t = u :: A -> f t = g u :: Bt Assumes that Bt ≡ Bu for this to make sense!

| Inject of proof * Constr.pconstructor * int * int

⊢ ci v = ci w :: Ind(args) -> ⊢ v = w :: T where T is the type of the n-th argument of ci, assuming they coincide

and proof = private {
p_lhs : Cc_plugin.Ccalgo.ATerm.t;
p_rhs : Cc_plugin.Ccalgo.ATerm.t;
p_rule : rule;
}
